#include "stdafx.h"
#include <windows.h>
#include <glut.h>   //Подключение библиотеки glut.h
#include <conio.h>

void init(void) 
      
{ 
//Выбрать фоновый (очищающий) цвет
glClearColor(1.0,1.0,1.0,0.0);// колір фону червоний/зелений/голубий колір в який очищується
//Установить проекцию
glMatrixMode(GL_PROJECTION);
glLoadIdentity(); 
glOrtho(0.0,1.0,0.0,1.0,-1.0,1.0); 
}


void display(void)
{
	//Очистить экран 
glClear(GL_COLOR_BUFFER_BIT);   

glColor3f(0.8,0.8,0.8);// Робимо вміст стінок ящика
glBegin(GL_POLYGON);
glVertex3f(0.75,0.55,0.0);

glColor3f(0.8,0.8,0.8);
glBegin(GL_POLYGON);

glVertex3f(0.75,0.35,0.0);

glEnd(); 

glColor3f(0.8,0.8,0.8);
glBegin(GL_POLYGON);

glVertex3f(0.87,0.55,0.0);

glVertex3f(0.85,0.31,0.0);
glVertex3f(0.87,0.31,0.0);
glEnd(); 


glColor3f(0.0,0.0,0.0);//чорний колір

glBegin(GL_POLYGON);//побудова ніжки ящика через полігон
	
   //вказуємо вершини полігона
    glVertex3f(0.805,0.31,0.0);
    glVertex3f(0.820,0.31,0.0);
    glVertex3f(0.820,0.20,0.0);
    glVertex3f(0.805,0.20,0.0);
    glVertex3f(0.805,0.31,0.0);
	
    glEnd();

glEnd(); 



    
//Не ждем. Начинаем выполнять буферизованные 
//команды OpenGL 
glFlush();
}  
//Установить начальные характеристики окна,
//открыть окно с заголовком «hello». 
//Зарегистрировать дисплейную функцию обратного вызова
//Войти в главный цикл
int main(int argc, char **argv) 
{
	
glutInit(&argc,argv);//встановлення початкових характеристик вікна
glutInitDisplayMode(GLUT_SINGLE|GLUT_RGB); 

glutInitWindowSize(1300,600);//Розмір вікна виводу результату  
glutInitWindowPosition(20,40);//Позиція вікна виводу результату
glutCreateWindow("my second program");//Назва вікна виводу результату
init(); 
glutDisplayFunc(display);
glutMainLoop(); //увійти до головного циклу 
return 0; 
}
